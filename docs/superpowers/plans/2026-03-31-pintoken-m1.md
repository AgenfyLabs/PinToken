# PinToken M1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working local LLM API proxy with cost tracking dashboard — install to first data visible in under 2 minutes.

**Architecture:** Single Node.js process serving proxy + API + static dashboard on port 7777. Proxy intercepts Anthropic/OpenAI requests, extracts usage from streaming SSE responses via side-channel parsing, calculates costs against pricing.json, stores in SQLite. Dashboard polls API every 5s.

**Tech Stack:** Node.js (ESM), better-sqlite3, nanoid, native HTTP module, vanilla HTML/CSS/JS

```javascript
// src/db/store.mjs
import Database from 'better-sqlite3';
import { mkdirSync } from 'node:fs';
import { dirname } from 'node:path';

const SCHEMA = `
CREATE TABLE IF NOT EXISTS requests (
  id            TEXT PRIMARY KEY,
  timestamp     TEXT NOT NULL,
  provider      TEXT NOT NULL,
  model         TEXT NOT NULL,
  input_tokens  INTEGER DEFAULT 0,
  output_tokens INTEGER DEFAULT 0,
  cache_read_tokens  INTEGER DEFAULT 0,
  cache_write_tokens INTEGER DEFAULT 0,
  cost_usd      REAL DEFAULT 0,
  baseline_cost_usd REAL DEFAULT 0,
  saved_usd     REAL DEFAULT 0,
  latency_ms    INTEGER DEFAULT 0,
  status_code   INTEGER DEFAULT 200
);

CREATE INDEX IF NOT EXISTS idx_timestamp ON requests(timestamp);
CREATE INDEX IF NOT EXISTS idx_provider ON requests(provider);
`;

/**
 * 创建数据存储实例
 * @param {string} dbPath - 数据库文件路径，默认 ~/.pintoken/data.db
 */
export function createStore(dbPath) {
  // 确保目录存在
  mkdirSync(dirname(dbPath), { recursive: true });

  const db = new Database(dbPath);
  db.pragma('journal_mode = WAL');
  db.exec(SCHEMA);

  // 预编译语句
  const insertStmt = db.prepare(`
    INSERT INTO requests (id, timestamp, provider, model, input_tokens, output_tokens,
      cache_read_tokens, cache_write_tokens, cost_usd, baseline_cost_usd, saved_usd,
      latency_ms, status_code)
    VALUES (@id, @timestamp, @provider, @model, @input_tokens, @output_tokens,
      @cache_read_tokens, @cache_write_tokens, @cost_usd, @baseline_cost_usd, @saved_usd,
      @latency_ms, @status_code)
  `);

  return {
    /** 插入一条请求记录 */
    insertRequest(record) {
      insertStmt.run(record);
    },

    /** 查询请求列表（分页 + 筛选） */
    getRequests({ provider, limit = 50, offset = 0 } = {}) {
      if (provider) {
        return db.prepare(
          'SELECT * FROM requests WHERE provider = ? ORDER BY timestamp DESC LIMIT ? OFFSET ?'
        ).all(provider, limit, offset);
      }
      return db.prepare(
        'SELECT * FROM requests ORDER BY timestamp DESC LIMIT ? OFFSET ?'
      ).all(limit, offset);
    },

    /** 获取今日摘要 */
    getSummary() {
      const today = new Date().toISOString().slice(0, 10); // YYYY-MM-DD
      const yesterday = new Date(Date.now() - 86400000).toISOString().slice(0, 10);

      const todayRow = db.prepare(`
        SELECT
          COALESCE(SUM(input_tokens + output_tokens), 0) as today_tokens,
          COALESCE(SUM(cost_usd), 0) as today_cost,
          COUNT(*) as today_requests
        FROM requests WHERE timestamp >= ?
      `).get(today);

      const yesterdayRow = db.prepare(`
        SELECT
          COALESCE(SUM(input_tokens + output_tokens), 0) as yesterday_tokens,
          COALESCE(SUM(cost_usd), 0) as yesterday_cost
        FROM requests WHERE timestamp >= ? AND timestamp < ?
      `).get(yesterday, today);

      const totalRow = db.prepare(`
        SELECT COALESCE(SUM(saved_usd), 0) as total_saved
        FROM requests
      `).get();

      return {
        today_tokens: todayRow.today_tokens,
        today_cost: todayRow.today_cost,
        today_requests: todayRow.today_requests,
        yesterday_tokens: yesterdayRow.yesterday_tokens,
        yesterday_cost: yesterdayRow.yesterday_cost,
        total_saved: totalRow.total_saved,
      };
    },

    /** 获取各 provider 统计 */
    getProviderStats() {
      return db.prepare(`
        SELECT
          provider,
          COUNT(*) as request_count,
          SUM(input_tokens + output_tokens) as total_tokens,
          SUM(cost_usd) as total_cost,
          SUM(saved_usd) as total_saved
        FROM requests
        GROUP BY provider
        ORDER BY total_cost DESC
      `).all();
    },

    /** 关闭数据库连接 */
    close() {
      db.close();
    },
  };
}

/** 默认数据库路径 */
export function getDefaultDbPath() {
  const home = process.env.HOME || process.env.USERPROFILE;
  return `${home}/.pintoken/data.db`;
}
```

```javascript
// src/pricing/calculator.mjs
import { readFileSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = dirname(fileURLToPath(import.meta.url));

// 加载定价数据
const pricingPath = join(__dirname, '../../data/pricing.json');
const PRICING = JSON.parse(readFileSync(pricingPath, 'utf-8'));

// 每个 provider 的最贵模型（用于 baseline 计算）
const BASELINE_MODELS = {
  anthropic: 'anthropic/claude-opus-4-6',
  openai: 'openai/gpt-4o',
};

/**
 * 获取指定 provider 的 baseline 模型 ID
 */
export function getBaselineModel(provider) {
  return BASELINE_MODELS[provider] || null;
}

/**
 * 计算一次 API 调用的费用
 * @returns {{ cost_usd, baseline_cost_usd, saved_usd, unknown }}
 */
export function calculateCost({ provider, model, input_tokens, output_tokens, cache_read_tokens, cache_write_tokens }) {
  // 构建 pricing key：provider/model
  const key = model.includes('/') ? model : `${provider}/${model}`;
  const pricing = PRICING[key];

  if (!pricing) {
    return { cost_usd: 0, baseline_cost_usd: 0, saved_usd: 0, unknown: true };
  }

  // 计算实际费用
  const cost_usd = computeCostFromPricing(pricing, input_tokens, output_tokens, cache_read_tokens, cache_write_tokens);

  // 计算 baseline 费用（同 provider 最贵模型）
  const baselineKey = BASELINE_MODELS[provider];
  const baselinePricing = baselineKey ? PRICING[baselineKey] : null;

  let baseline_cost_usd = 0;
  if (baselinePricing && baselineKey !== key) {
    baseline_cost_usd = computeCostFromPricing(baselinePricing, input_tokens, output_tokens, cache_read_tokens, cache_write_tokens);
  } else {
    // 已经是最贵模型，baseline = actual
    baseline_cost_usd = cost_usd;
  }

  const saved_usd = Math.max(0, baseline_cost_usd - cost_usd);

  return { cost_usd, baseline_cost_usd, saved_usd, unknown: false };
}

/**
 * 根据定价计算费用
 */
function computeCostFromPricing(pricing, input_tokens, output_tokens, cache_read_tokens, cache_write_tokens) {
  let cost = 0;
  cost += (input_tokens / 1_000_000) * pricing.input_per_1m;
  cost += (output_tokens / 1_000_000) * pricing.output_per_1m;
  if (cache_read_tokens && pricing.cache_read_per_1m) {
    cost += (cache_read_tokens / 1_000_000) * pricing.cache_read_per_1m;
  }
  if (cache_write_tokens && pricing.cache_write_per_1m) {
    cost += (cache_write_tokens / 1_000_000) * pricing.cache_write_per_1m;
  }
  return cost;
}
```

```javascript
/**
 * 解析 Anthropic SSE 流数据，提取 usage 信息
 * @param {string} rawSSE - 完整的 SSE 文本（所有 chunk 拼接）
 * @returns {{ model, input_tokens, output_tokens, cache_read_tokens, cache_write_tokens } | null}
 */
export function parseAnthropicSSE(rawSSE) {
  let model = null;
  let input_tokens = 0;
  let output_tokens = 0;
  let cache_read_tokens = 0;
  let cache_write_tokens = 0;

  // 按事件分割
  const events = rawSSE.split('\n\n').filter(Boolean);

  for (const event of events) {
    const dataLine = event.split('\n').find(l => l.startsWith('data: '));
    if (!dataLine) continue;

    try {
      const data = JSON.parse(dataLine.slice(6));

      if (data.type === 'message_start' && data.message) {
        model = data.message.model;
        if (data.message.usage) {
          input_tokens = data.message.usage.input_tokens || 0;
          cache_read_tokens = data.message.usage.cache_read_input_tokens || 0;
          cache_write_tokens = data.message.usage.cache_creation_input_tokens || 0;
        }
      }

      if (data.type === 'message_delta' && data.usage) {
        output_tokens = data.usage.output_tokens || 0;
      }
    } catch {
      // 解析失败，跳过
    }
  }

  if (!model) return null;

  return { model, input_tokens, output_tokens, cache_read_tokens, cache_write_tokens };
}

/**
 * 解析 OpenAI SSE 流数据，提取 usage 信息
 * @param {string} rawSSE - 完整的 SSE 文本
 * @returns {{ model, input_tokens, output_tokens } | null}
 */
export function parseOpenAISSE(rawSSE) {
  let model = null;
  let input_tokens = 0;
  let output_tokens = 0;

  const lines = rawSSE.split('\n').filter(l => l.startsWith('data: '));

  for (const line of lines) {
    const payload = line.slice(6).trim();
    if (payload === '[DONE]') continue;

    try {
      const data = JSON.parse(payload);
      if (data.model) model = data.model;
      if (data.usage) {

        output_tokens = data.usage.completion_tokens || 0;
      }
    } catch {
      // 解析失败，跳过
    }
  }

  if (!model || (input_tokens === 0 && output_tokens === 0)) return null;

  return { model, input_tokens, output_tokens, cache_read_tokens: 0, cache_write_tokens: 0 };
}
```

```javascript
// src/proxy/anthropic.mjs
import https from 'node:https';
import { nanoid } from 'nanoid';
import { parseAnthropicSSE, parseAnthropicResponse } from './streaming.mjs';
import { calculateCost } from '../pricing/calculator.mjs';

const ANTHROPIC_HOST = 'api.anthropic.com';

/**
 * 处理 Anthropic 代理请求
 * @param {http.IncomingMessage} clientReq
 * @param {http.ServerResponse} clientRes
 * @param {object} store - SQLite 存储实例
 * @param {function} onLog - 日志回调
 */
export function handleAnthropic(clientReq, clientRes, store, onLog) {
  const startTime = Date.now();

  // 去掉 /anthropic 前缀
  const path = clientReq.url.replace(/^\/anthropic/, '');

  // 读取客户端请求 body
  const bodyChunks = [];
  clientReq.on('data', chunk => bodyChunks.push(chunk));
  clientReq.on('end', () => {
    const body = Buffer.concat(bodyChunks);

    // 判断是否 streaming
    let isStreaming = false;
    try {
      const parsed = JSON.parse(body.toString());
      isStreaming = parsed.stream === true;
    } catch {}

    // 构建上游请求
    const options = {
      hostname: ANTHROPIC_HOST,
      port: 443,
      path,
      method: clientReq.method,
      headers: { ...clientReq.headers, host: ANTHROPIC_HOST },
    };

    const proxyReq = https.request(options, (proxyRes) => {
      // 复制响应头
      clientRes.writeHead(proxyRes.statusCode, proxyRes.headers);

      if (isStreaming) {
        // Streaming：实时 pipe + 旁路收集
        const sseChunks = [];
        proxyRes.on('data', (chunk) => {
          clientRes.write(chunk); // 实时透传
          sseChunks.push(chunk); // 旁路收集
        });
        proxyRes.on('end', () => {
          clientRes.end();
          // 解析并记录
          const raw = Buffer.concat(sseChunks).toString();
          const usage = parseAnthropicSSE(raw);
          if (usage) {
            recordRequest(store, usage, 'anthropic', proxyRes.statusCode, startTime, onLog);
          }
        });
      } else {
        // 非 streaming：收集完整 body
        const resChunks = [];
        proxyRes.on('data', chunk => resChunks.push(chunk));
        proxyRes.on('end', () => {
          const resBody = Buffer.concat(resChunks);
          clientRes.end(resBody);
          // 解析并记录
          const usage = parseAnthropicResponse(resBody.toString());
          if (usage) {
            recordRequest(store, usage, 'anthropic', proxyRes.statusCode, startTime, onLog);
          }
        });
      }
    });

    proxyReq.on('error', (err) => {
      clientRes.writeHead(502);
      clientRes.end(JSON.stringify({ error: 'Proxy error', message: err.message }));
    });

    proxyReq.write(body);
    proxyReq.end();
  });
}

/**
 * 记录请求到 SQLite
 */
function recordRequest(store, usage, provider, statusCode, startTime, onLog) {
  const cost = calculateCost({
    provider,
    model: usage.model,
    input_tokens: usage.input_tokens,
    output_tokens: usage.output_tokens,
    cache_read_tokens: usage.cache_read_tokens,
    cache_write_tokens: usage.cache_write_tokens,
  });

  const record = {
    id: nanoid(),
    timestamp: new Date().toISOString(),
    provider,
    model: usage.model,
    input_tokens: usage.input_tokens,
    output_tokens: usage.output_tokens,
    cache_read_tokens: usage.cache_read_tokens,
    cache_write_tokens: usage.cache_write_tokens,
    cost_usd: cost.cost_usd,
    baseline_cost_usd: cost.baseline_cost_usd,
    saved_usd: cost.saved_usd,
    latency_ms: Date.now() - startTime,
    status_code: statusCode,
  };

  store.insertRequest(record);
  onLog(record);
}
```

```javascript
// src/api/routes.mjs
import { URL } from 'node:url';

/**
 * 处理 /api/* 请求
 * @param {http.IncomingMessage} req
 * @param {http.ServerResponse} res
 * @param {object} store - SQLite 存储实例
 * @param {number} startedAt - 服务器启动时间戳（用于计算会话时长）
 */
export function handleAPI(req, res, store, startedAt) {
  const url = new URL(req.url, `http://${req.headers.host}`);
  const path = url.pathname;

  // CORS 头（Dashboard 可能从不同端口访问）
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Content-Type', 'application/json');

  try {
    if (path === '/api/summary') {
      const summary = store.getSummary();
      // 计算会话时长（秒）
      summary.session_seconds = Math.floor((Date.now() - startedAt) / 1000);
      res.end(JSON.stringify(summary));
      return;
    }

    if (path === '/api/requests') {
      const provider = url.searchParams.get('provider') || undefined;
      const limit = parseInt(url.searchParams.get('limit') || '50', 10);
      const offset = parseInt(url.searchParams.get('offset') || '0', 10);
      const rows = store.getRequests({ provider, limit, offset });
      res.end(JSON.stringify(rows));
      return;
    }

    if (path === '/api/providers') {
      const stats = store.getProviderStats();
      res.end(JSON.stringify(stats));
      return;
    }

    // 未知 API 路径
    res.writeHead(404);
    res.end(JSON.stringify({ error: 'Not found' }));
  } catch (err) {
    res.writeHead(500);
    res.end(JSON.stringify({ error: err.message }));
  }
}
```

```javascript
// src/proxy/server.mjs
import http from 'node:http';
import { readFileSync, existsSync, statSync } from 'node:fs';
import { join, extname, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { handleAnthropic } from './anthropic.mjs';
import { handleOpenAI } from './openai.mjs';
import { handleAPI } from '../api/routes.mjs';
import { createStore, getDefaultDbPath } from '../db/store.mjs';

const __dirname = dirname(fileURLToPath(import.meta.url));
const DASHBOARD_DIR = join(__dirname, '../../dashboard');

// MIME 类型映射
const MIME_TYPES = {
  '.html': 'text/html',
  '.css': 'text/css',
  '.js': 'application/javascript',
  '.json': 'application/json',
  '.png': 'image/png',
  '.svg': 'image/svg+xml',
  '.ico': 'image/x-icon',
};

/**
 * 启动 PinToken 服务器
 * @param {{ port?: number, dbPath?: string, onLog?: function }} options
 * @returns {Promise<http.Server>}
 */
export function startServer({ port = 7777, dbPath, onLog } = {}) {
  const store = createStore(dbPath || getDefaultDbPath());
  const startedAt = Date.now();

  // 默认日志回调
  const log = onLog || ((record) => {
    const time = new Date().toLocaleTimeString('en-US', { hour12: false });
    const tokens = `${record.input_tokens.toLocaleString()} in  ${record.output_tokens.toLocaleString()} out`;
    const cost = `$${record.cost_usd.toFixed(4)}`;
    const saved = record.saved_usd > 0 ? `saved $${record.saved_usd.toFixed(3)}` : '';
    console.log(`[${time}] ${record.model.padEnd(25)} | ${tokens.padEnd(20)} | ${cost.padEnd(10)} | ${saved}`);
  });

  const server = http.createServer((req, res) => {
    const url = req.url;

    // 代理路由
    if (url.startsWith('/anthropic/')) {
      handleAnthropic(req, res, store, log);
      return;
    }

    if (url.startsWith('/openai/')) {
      handleOpenAI(req, res, store, log);
      return;
    }

    // API 路由
    if (url.startsWith('/api/')) {
      handleAPI(req, res, store, startedAt);
      return;
    }

    // 静态文件路由（Dashboard）
    serveDashboard(req, res);
  });

  return new Promise((resolve) => {
    server.listen(port, () => {
      resolve({ server, store, port });
    });
  });
}

/**
 * 提供 Dashboard 静态文件
 */
function serveDashboard(req, res) {
  let filePath = req.url === '/' ? '/index.html' : req.url;
  // 安全检查：防止目录遍历
  filePath = filePath.replace(/\.\./g, '');
  const fullPath = join(DASHBOARD_DIR, filePath);

  if (!existsSync(fullPath) || statSync(fullPath).isDirectory()) {
    // SPA fallback
    const indexPath = join(DASHBOARD_DIR, 'index.html');
    if (existsSync(indexPath)) {
      res.writeHead(200, { 'Content-Type': 'text/html' });
      res.end(readFileSync(indexPath));
    } else {
      res.writeHead(404);
      res.end('Dashboard not found');
    }
    return;
  }

  const ext = extname(fullPath);
  const mime = MIME_TYPES[ext] || 'application/octet-stream';
  res.writeHead(200, { 'Content-Type': mime });
  res.end(readFileSync(fullPath));
}
```

```javascript
// src/setup/shell.mjs
// Shell 检测 + profile 写入（参考 pintoken-setup-reference.md）
import { readFileSync, writeFileSync, appendFileSync, existsSync, mkdirSync } from 'node:fs';
import { join, basename, dirname } from 'node:path';
import { homedir, platform } from 'node:os';
import { execSync } from 'node:child_process';

const MARKER = '# Added by PinToken';

/** 检测用户当前 shell */
export function detectShell() {
  if (process.env.CLAUDE_CODE_SHELL) {
    return normalizeShellName(process.env.CLAUDE_CODE_SHELL);
  }
  try {
    const ppid = process.ppid;
    const parentName = execSync(`ps -o comm= -p ${ppid}`, {
      encoding: 'utf-8', timeout: 3000,
    }).trim();
    const shell = normalizeShellName(parentName);
    if (shell) return shell;
  } catch {}
  if (process.env.SHELL) return normalizeShellName(process.env.SHELL);
  return 'bash';
}

function normalizeShellName(raw) {
  const name = basename(raw).replace(/^-/, '');
  if (name.includes('zsh')) return 'zsh';
  if (name.includes('bash')) return 'bash';
  if (name.includes('fish')) return 'fish';
  return name || null;
}

/** 获取 shell profile 路径 */
export function getShellProfilePath(shell) {
  const home = homedir();
  const isMac = platform() === 'darwin';
  const candidates = {
    zsh: [join(home, '.zshrc'), join(home, '.zprofile')],
    bash: isMac
      ? [join(home, '.bash_profile'), join(home, '.bashrc'), join(home, '.profile')]
      : [join(home, '.bashrc'), join(home, '.bash_profile'), join(home, '.profile')],
    fish: [join(home, '.config', 'fish', 'config.fish')],
  };
  const paths = candidates[shell] || candidates.bash;
  return paths.find(p => existsSync(p)) || paths[0];
}

/** 幂等写入 shell profile 环境变量 */
export function appendEnvToProfile(profilePath, key, value) {
  mkdirSync(dirname(profilePath), { recursive: true });

  const exportLine = `export ${key}="${value}"`;
  const markedLine = `${exportLine}  ${MARKER}`;

  if (existsSync(profilePath)) {
    const content = readFileSync(profilePath, 'utf-8');
    const regex = new RegExp(`^export ${escapeRegExp(key)}=.*${escapeRegExp(MARKER)}$`, 'm');

    if (regex.test(content)) {
      writeFileSync(profilePath, content.replace(regex, markedLine));
      return 'updated';
    }

    if (new RegExp(`^export ${escapeRegExp(key)}=`, 'm').test(content)) {
      return 'skipped';
    }
  }

  appendFileSync(profilePath, `\n${markedLine}\n`);
  return 'added';
}

/** 检查是否已配置过 */
export function isConfigured(profilePath) {
  if (!existsSync(profilePath)) return false;
  const content = readFileSync(profilePath, 'utf-8');
  return content.includes(MARKER);
}

function escapeRegExp(s) {
  return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}
```

```javascript
import { readFileSync, writeFileSync, existsSync, mkdirSync } from 'node:fs';

import { homedir } from 'node:os';

const SETTINGS_PATH = join(homedir(), '.claude', 'settings.json');

/** 设置 Claude Code 的 ANTHROPIC_BASE_URL */
export function setClaudeBaseUrl(baseUrl) {
  mkdirSync(dirname(SETTINGS_PATH), { recursive: true });

  let settings = {};
  if (existsSync(SETTINGS_PATH)) {
    try { settings = JSON.parse(readFileSync(SETTINGS_PATH, 'utf-8')); } catch {}
  }
  settings.env = settings.env || {};
  settings.env.ANTHROPIC_BASE_URL = baseUrl;
  writeFileSync(SETTINGS_PATH, JSON.stringify(settings, null, 2) + '\n');
  return SETTINGS_PATH;
}
```

```javascript
// src/setup/index.mjs
// setup 主流程编排
import { detectShell, getShellProfilePath, appendEnvToProfile, isConfigured } from './shell.mjs';
import { setClaudeBaseUrl } from './claude.mjs';
import { openBrowser } from './browser.mjs';
import { existsSync, readFileSync, writeFileSync } from 'node:fs';
import { resolve } from 'node:path';

const ANTHROPIC_BASE_URL = 'http://localhost:7777/anthropic';
const OPENAI_BASE_URL = 'http://localhost:7777/openai';

/**
 * 执行 setup 流程
 * @returns {boolean} 是否执行了新配置（false = 已配置过，跳过）
 */
export function runSetup() {
  const shell = detectShell();
  const profilePath = getShellProfilePath(shell);
  const alreadyConfigured = isConfigured(profilePath);

  if (alreadyConfigured) {
    console.log('  已检测到 PinToken 配置，跳过配置步骤');
    return false;
  }

  console.log(`  检测到 shell: ${shell}`);
  console.log(`  Profile: ${profilePath}`);
  console.log('');

  // 1. 写入 shell profile
  const r1 = appendEnvToProfile(profilePath, 'ANTHROPIC_BASE_URL', ANTHROPIC_BASE_URL);
  console.log(`  ANTHROPIC_BASE_URL → ${profilePath} (${r1})`);

  const r2 = appendEnvToProfile(profilePath, 'OPENAI_BASE_URL', OPENAI_BASE_URL);
  console.log(`  OPENAI_BASE_URL → ${profilePath} (${r2})`);

  // 2. 写入 Claude Code settings.json
  const settingsPath = setClaudeBaseUrl(ANTHROPIC_BASE_URL);
  console.log(`  Claude Code → ${settingsPath}`);

  // 3. 写入 .env（如果当前目录有的话）
  const envPath = resolve('.env');
  if (existsSync(envPath)) {
    upsertEnv(envPath, 'ANTHROPIC_BASE_URL', ANTHROPIC_BASE_URL);
    upsertEnv(envPath, 'OPENAI_BASE_URL', OPENAI_BASE_URL);
    console.log(`  .env → ${envPath}`);
  }

  console.log('');
  console.log(`  配置完成! 重启 shell 或运行: source ${profilePath}`);
  return true;
}

/** 简易 .env upsert */
function upsertEnv(envPath, key, value) {
  let content = readFileSync(envPath, 'utf-8');
  const regex = new RegExp(`^${key}=.*$`, 'm');
  if (regex.test(content)) {
    content = content.replace(regex, `${key}=${value}`);
  } else {
    content += `\n${key}=${value}\n`;
  }
  writeFileSync(envPath, content);
}
```
